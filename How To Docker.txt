1. Install Docker (When using Windows it needs to be Professional)
2. git clone https://github.com/HansBambel/nextWordPrediction.git
# Creates an image (the requirements) --> only needs to be done once
3. docker build --tag=transformers .

# Helpful Docker commands:
	a. docker image ls (lists all "installed" images)
	b. docker ps -a (shows all containers)
	c. docker rm <container-name> (container-name is at the end of docker ps command)
	d. docker run <image-name> (-d =detached/background, -it =interactive shell, --rm =removes container after exit, -m 32g =allows the container to use 32gb of RAM)
	e. docker exec -it <container-name> /bin/bash (enter running container)
	f. docker cp . <container-name>:/gpt2 (copies files from host to container)
	g. docker stop <container-name>	(stops the container)
	h. docker container prune (removes all stopped containers)

# creates an instance of the image
4. docker run -it -d transformers
5. get container name from "docker ps" command
6. Copy files from host to container: "docker cp . zealous_ptolemy:/gpt2"
7. enter running container again: "docker exec -it <container-name> /bin/bash"
8. run your script "python probabilities.py"
